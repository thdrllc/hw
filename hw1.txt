	#include <stdio.h>
	#include <stdlib.h>
	#include <locale.h>
	int main()
	{
	setlocale(LC_ALL, "Rus");
	  int i, j,max;
	  int sumstr[8];
	  int massiv[8][8]; 
	  max=0; 
	  	  srand(time(NULL));
	  for ( i = 0; i < 8; i++ ) {
	   sumstr[i]=0;
		for ( j = 0; j < 8; j++ ){
		  massiv[i][j] = 20+rand()%80; 
		  printf( "%d ",massiv[i][j] );
		  sumstr[i]+=massiv[i][j];
	  }
	  	printf( "Сумма элементов строки %d\n", sumstr[i] );
		if(sumstr[i] > max)
            {
                max = sumstr[i];
            }
	  }
  printf("Максимальная сумма элементов строки=%d", max);
}
